int bar(int x) { return x / 2; }

struct A {
    int a(int x)
    {
        if (x % 2 == 0) {
            return x;
        }
        else if (x > 2) {
            return bar(x);
        }
        else if (x == 123) {
            auto f = [](auto &v) { return v + 1; };

            return f(x);
        }
        else {
            return 0;
        }
    }
};

int tmain()
{

        return x;
    }
    else if (x == 200) {
        return baz(x);

    return 0;
}
