    return 1000 * p;
}

template <typename T> T m2(T p) { return m3<T>(p); }

    return std::string{"m2"} + p;
}

{
    return m4<unsigned long>(p);
}

template <> [[maybe_unused]] std::string m1<std::string>(std::string p)
{
    return m2<std::string>(p);
}

int main()
{
    m1<float>(2.2);

    m1<unsigned long>(100);

    m1<std::string>("main");

    return m1<int>(0);
}
}
}
