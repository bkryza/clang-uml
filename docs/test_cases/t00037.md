    double x;

};

    struct {
        double t;
        double x;
        double y;
        double z;
    } dimensions;

    struct {
        int len;
        int flags;
    }

    bars[LENGTH];

private:
    struct {
        double c{1.0};

    } units;

    S s[4][3][2];
};

    A()
    {
        st.dimensions.t = -1.0;
        st.dimensions.x = 1.0;
        st.dimensions.y = 1.0;
        st.dimensions.z = 1.0;
    }

    ST st;
};
